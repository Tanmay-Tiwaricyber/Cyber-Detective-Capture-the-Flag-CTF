# Ghost's encrypted message
message = "UDVG{pybg-arfggrk}"
key = 13

def decrypt(message, key):
    decrypted = ""
    for char in message:
        if char.isalpha():
            shift = 65 if char.isupper() else 97
            decrypted += chr((ord(char) - shift + key) % 26 + shift)
        else:
            decrypted += char
    return decrypted

flag = decrypt(message, key)
print("Flag: " + flag)  # <--- Solve this!

Clue: The message is encrypted with a Caesar cipher. Can you decrypt it?
